import re

Asyncio

Asinxron dasturlash — bir vaqtda ko'p vazifa bajarish. Telegram botlar uchun muhim.

python
import asyncio

async def salom(ism):
    print(f"Salom {ism}")
    await asyncio.sleep(1)    # blokirovka qilmaydi
    print(f"{ism} tugadi")

# Ishga tushirish
asyncio.run(salom("Ali"))

# Bir vaqtda bajarish (gather)
async def asosiy():
    await asyncio.gather(
        salom("Ali"),
        salom("Vali"),
        salom("Soli")
    )

# Task yaratish
async def asosiy():
    t1 = asyncio.create_task(salom("Ali"))
    t2 = asyncio.create_task(salom("Vali"))
    await t1
    await t2

# Timeout
async def asosiy():
    try:
        await asyncio.wait_for(salom("Ali"), timeout=2.0)
    except asyncio.TimeoutError:
        print("Vaqt tugadi")

for k, g in itertools.groupby([1,1,2,2,3]):
